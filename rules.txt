PENJELASAN STRUKTUR FOLDER
TEMANMASAK
│
├── node_modules/         #Pustaka dan dependensi proyek
├── public/               #File statis (index.html, favicon, gambar)
└── src/                  #Pusat kode sumber aplikasi Anda
    ├── assets/           #Gambar, font, atau ikon lokal
    ├── components/       #Komponen global yang dapat digunakan ulang (Button, Modal)
    ├── features/         #Logika spesifik per fitur (dashboard, auth, profile)
    │   ├── auth/         
    │   │   ├── components/   #Komponen khusus untuk fitur auth
    │   │   ├── services/     #API/fetch khusus auth
    │   │   └── index.js      
    ├── hooks/            #Custom React Hooks (useFetch, dll)
    ├── layouts/          #Kerangka halaman utama (Navbar, Sidebar, Footer)
    ├── pages/            #Halaman utama yang memanggil komponen dan layout
    ├── services/         #Konfigurasi dan panggilan API global (Axios/Fetch)
    ├── styles/           #File CSS global atau konfigurasi tema (Tailwind/Sass)
    ├── utils/            #Fungsi pembantu (format tanggal, helper matematika)
    ├── App.js            #Komponen utama yang menjadi pusat routing
    └── index.js          #Titik masuk aplikasi ke DOM
